== Abfrage

Erhöhen aller Strafen, die über dem Durchschnitt liegen, um 20%

[source, sql]
----
update PENALTIES
set AMOUNT = AMOUNT * 1.2
where AMOUNT > (select avg(AMOUNT) from PENALTIES);
commit;
----
