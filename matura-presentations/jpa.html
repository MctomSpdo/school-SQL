3. JPA Entity

@Table(name = "JPA_CAR")
@Entity
@TableGenerator(name = "idtab", initialValue = 0, allocationSize = 10)
public class Car {
    @Id
    @GeneratedValue(strategy = GenerationType.TABLE, generator = "idtab")
    Long id;

    String name;
}
